        // Función de predicción

        function predecirPrecio(area, bedrooms, bathrooms, barrio, propertyType) {

            // Precio base por m²

            let precioM2 = preciosM2[barrio] || 4000000;

            

            // Ajustes por tipo de propiedad

            if (propertyType === 'Penthouse') precioM2 *= 1.3;

            if (propertyType === 'Casa') precioM2 *= 0.95;

            if (propertyType === 'Loft') precioM2 *= 1.1;

            

            // Cálculo precio base

            let precioBase = precioM2 * area;

            

            // Ajustes por características

            precioBase += bedrooms * 15000000; // Bonus por habitación

            precioBase += bathrooms * 10000000; // Bonus por baño

            

            // Ajuste logarítmico para simular el modelo real

            // (el modelo real usa transformación log)

            let factor_log = Math.log(area) / Math.log(80); // Normalizado a 80m²

            precioBase *= factor_log;

            

            return precioBase;

        }

        

        // Formatear número como moneda

        function formatCurrency(num) {

            return new Intl.NumberFormat('es-CO', {

                style: 'currency',

                currency: 'COP',

                minimumFractionDigits: 0,

                maximumFractionDigits: 0

            }).format(num);

        }

            const bedrooms = parseInt(document.getElementById('bedrooms').value);

            const bathrooms = parseInt(document.getElementById('bathrooms').value);

            const propertyType = document.getElementById('propertyType').value;

            

            // Predicción

            const precioActual = predecirPrecio(area, bedrooms, bathrooms, barrio, propertyType);

            const factor = factores[barrio] || 1.11;

            const precio2025 = precioActual * factor;

            const incrementoPct = (factor - 1) * 100;

            

            // Mostrar resultados

            document.getElementById('precio2025').textContent = formatCurrency(precio2025);

                incrementoEl.className = 'increment high';

            } else if (incrementoPct > 10) {

                incrementoEl.className = 'increment medium';

            } else {

                incrementoEl.className = 'increment low';

            }
